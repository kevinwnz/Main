<h2 class="centxt"> Create a text file and save it to a local computer using JavaScript. </h2>
       <h4 style="color: red;" class="centxt"> Enter the file content:- </h4>
       <textarea style="margin-left: 40%; margin-right: 40%;" cols="50" rows="10"> </textarea>
       <br/>
       <button style="margin-left: 40%; margin-right: 40%;" onclick = "downloadFile()"> save File </button><p> </p>
       <p> ------------------------------------------------------------------------------------------------------------- </p>
       <script>
          const downloadFile = () => {
             const link = document.createElement("a");
             const content = document.querySelector("textarea").value;
             const file = new Blob([content], { type: 'text/plain' });
             link.href = URL.createObjectURL(file);
             link.download = "sample.txt";
             link.click();
             URL.revokeObjectURL(link.href);
          };
       </script>
